onEvent

Logs events in default console color

Args:

  • x {String}: message to log
onEvent = (x) ->
    unless quiet
        console.log "   #{x}"

onStep

Logs steps in blue

Args:

  • x {String}: message to log
onStep = (x) ->
    unless quiet
        console.log "#{x}".blue

onComplete

Logs successful process completions in green

Args:

  • x {String}: message to log
onComplete = (x) ->
    console.log "#{x}".green

onError

Logs errors in red

Args:

  • x {String}: message to log
onError = (x) ->
    console.log "!!! Error: #{x} !!!".red
